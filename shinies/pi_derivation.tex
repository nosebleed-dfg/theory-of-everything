% -- Abstract -----------------------------------------------------------------
\begin{abstract}
We derive $\pi$ from the single axiom $x^2 = x + 1$.
The axiom produces two roots, $\varphi = (1+\sqrt{5})/2$ and
$\psi = (1-\sqrt{5})/2 = -1/\varphi$, and forces three structural
constants: $\chi = 2$, $d = 3$, $p = 5$.
We prove algebraically that
$\cos(\pi/5) = \varphi/2 = \varphi/\chi$, whence
$\pi = p \cdot \arccos(\varphi/\chi)$ exactly.
We then show that the Pythagorean fold chain
$(1,1) \to (1,2) \to (2,5) \to (5,29)$ produces
the primes $\{23, 29\}$ which, together with $\{\chi, d, p\}$,
yield $640320 = 2^{\chi d} \cdot d \cdot p \cdot 23 \cdot 29$.
This integer is the base of the Chudnovsky series for $1/\pi$.
Every constant in the derivation traces to the axiom.
Every step terminates.
\end{abstract}

\section{The Axiom and Its Consequences}\label{sec:axiom}
% =============================================================================

\begin{definition}[The Axiom]\label{def:axiom}
The axiom is:
\begin{equation}\label{eq:axiom}
  \boxed{\; x^2 = x + 1 \;}
\end{equation}
This is equivalent to $x^2 - x - 1 = 0$.
\end{definition}

\begin{theorem}[Roots]\label{thm:roots}
The axiom has exactly two roots:
\begin{align}
  \varphi &= \frac{1 + \sqrt{5}}{2} \approx 1.618
  \quad\text{(the positive half),} \label{eq:phi} \\[4pt]
  \psi &= \frac{1 - \sqrt{5}}{2} \approx -0.618
  \quad\text{(the negative half).} \label{eq:psi}
\end{align}
\end{theorem}

\begin{proof}
The quadratic formula applied to $x^2 - x - 1 = 0$ gives
$x = (1 \pm \sqrt{1+4})/2 = (1 \pm \sqrt{5})/2$.
\end{proof}

\begin{proposition}[Root Identities]\label{prop:root-identities}
The roots satisfy:
\begin{enumerate}[nosep]
  \item $\varphi + \psi = 1$ \quad(sum of roots $= -(-1)/1$),
  \item $\varphi \cdot \psi = -1$ \quad(product of roots $= -1/1$),
  \item $\psi = -1/\varphi$ \quad(follows from (2)),
  \item $\varphi - \psi = \sqrt{5}$ \quad(follows from the definitions).
\end{enumerate}
\end{proposition}

\begin{proof}
By Vieta's formulas for $x^2 - x - 1 = 0$, the sum of roots is $1$ and
the product is $-1$. Then $\psi = -1/\varphi$ is immediate. Finally,
$\varphi - \psi = \frac{1+\sqrt{5}}{2} - \frac{1-\sqrt{5}}{2} = \sqrt{5}$.
\end{proof}

\section{The Forced Constants}\label{sec:constants}
% =============================================================================

\begin{definition}[Chi]\label{def:chi}
Define
\begin{equation}\label{eq:chi}
  \chi \;:=\; \varphi + \psi + 1 \;=\; 1 + 1 \;=\; 2.
\end{equation}
Equivalently, $\chi = \varphi - \psi + 1 = \sqrt{5} + 1 - (\sqrt{5} - 1) + 1$
fails; more directly, $\chi$ is forced as the smallest integer exceeding
both roots in absolute value:
$\lceil \varphi \rceil = 2$ and $\lceil |\psi|^{-1} \rceil = 2$.
The axiom's discriminant is $\Delta = 1 + 4 = 5 = \chi^2 + 1$.
\end{definition}

\begin{theorem}[Forced Triple]\label{thm:forced-triple}
From $\chi = 2$, the following constants are uniquely determined:
\begin{align}
  d &= \chi^2 - 1 = 3, \label{eq:d} \\
  p &= \chi^2 + 1 = 5. \label{eq:p}
\end{align}
\end{theorem}

\begin{proof}
$\chi^2 = 4$. Then $d = 4 - 1 = 3$ and $p = 4 + 1 = 5$.
\end{proof}

\begin{remark}[Identification]\label{rem:identification}
In the dodecahedron $\{p, d\} = \{5, 3\}$:
$p = 5$ is the number of sides per face (pentagonal faces),
$d = 3$ is the vertex degree (three faces meet at each vertex),
and $\chi = 2$ is the Euler characteristic ($V - E + F = 20 - 30 + 12 = 2$).
The Schl\"afli symbol is determined by $(\chi, d, p)$ alone.
\end{remark}

\section{The Fold Chain}\label{sec:folds}
% =============================================================================

\begin{definition}[Pythagorean Fold]\label{def:fold}
A \emph{Pythagorean fold} is the map $(a, b) \mapsto (b,\, a^2 + b^2)$.
It takes a pair and produces a new pair whose second component is the
Pythagorean sum of the inputs.
\end{definition}

\begin{theorem}[The Fold Chain]\label{thm:fold-chain}
Starting from $(1, 1)$, the fold chain is:
\begin{equation}\label{eq:fold-chain}
  (1, 1) \;\xrightarrow{\;\text{fold}\;}
  (1, 2) \;\xrightarrow{\;\text{fold}\;}
  (2, 5) \;\xrightarrow{\;\text{fold}\;}
  (5, 29).
\end{equation}
Reading the second coordinates: $1, 2, 5, 29$.
These are $1$, $\chi$, $p$, and $29$.
\end{theorem}

\begin{proof}
\begin{align*}
  (1, 1) &\mapsto (1,\; 1^2 + 1^2) = (1, 2), \\
  (1, 2) &\mapsto (2,\; 1^2 + 2^2) = (2, 5), \\
  (2, 5) &\mapsto (5,\; 2^2 + 5^2) = (5, 29). \qedhere
\end{align*}
\end{proof}

\begin{theorem}[Backward and Forward Folds]\label{thm:23-29}
Two additional constants emerge from $\chi$ and $p$:
\begin{align}
  23 &= p^2 - \chi = 25 - 2 = 23
  \quad\text{(backward fold),} \label{eq:23} \\
  29 &= \chi^2 + p^2 = 4 + 25 = 29
  \quad\text{(Pythagorean fold).} \label{eq:29}
\end{align}
Both $23$ and $29$ are prime.
\end{theorem}

\begin{proof}
Direct computation. Primality: $23$ is not divisible by $2, 3$
(and $\lfloor\sqrt{23}\rfloor = 4$), so $23$ is prime.
$29$ is not divisible by $2, 3, 5$ (and $\lfloor\sqrt{29}\rfloor = 5$),
so $29$ is prime.
\end{proof}

\section{The Algebraic Proof That $\cos(\pi/5) = \varphi/2$}
\label{sec:cos-proof}
% =============================================================================

This is the central algebraic step. We prove that $\varphi/2$ is a root
of the minimal polynomial of $\cos(\pi/5)$ by purely algebraic means,
then identify the correct root.

\begin{lemma}[Minimal Polynomial of $\cos(\pi/5)$]\label{lem:cheby}
$\cos(\pi/5)$ is a root of the polynomial
\begin{equation}\label{eq:cheby-poly}
  16t^4 - 12t^2 + 1 = 0.
\end{equation}
\end{lemma}

\begin{proof}
Let $\theta = \pi/5$, so $5\theta = \pi$ and $\cos(5\theta) = \cos\pi = -1$.
We expand $\cos(5\theta)$ in terms of $c = \cos\theta$ using the
Chebyshev identity:
\begin{equation}\label{eq:cos5}
  \cos(5\theta) = 16c^5 - 20c^3 + 5c.
\end{equation}
(This follows from the recurrence $T_n(c) = 2c\,T_{n-1}(c) - T_{n-2}(c)$
with $T_0 = 1$, $T_1 = c$, giving $T_2 = 2c^2 - 1$,
$T_3 = 4c^3 - 3c$, $T_4 = 8c^4 - 8c^2 + 1$,
$T_5 = 16c^5 - 20c^3 + 5c$.)

Setting $\cos(5\theta) = -1$:
\begin{equation}\label{eq:cos5-eq}
  16c^5 - 20c^3 + 5c + 1 = 0.
\end{equation}
We factor out the known root $c = \cos\pi = -1$ (since $\theta = \pi$
also satisfies $5\theta = 5\pi$ and $\cos(5\pi) = -1$). Indeed,
$(c + 1)$ divides the left side:
\[
  16c^5 - 20c^3 + 5c + 1 = (c + 1)(16c^4 - 16c^3 - 4c^2 + 4c + 1).
\]
(Verification: expanding $(c+1)(16c^4 - 16c^3 - 4c^2 + 4c + 1)$
gives $16c^5 - 16c^4 - 4c^3 + 4c^2 + c + 16c^4 - 16c^3 - 4c^2 + 4c + 1
= 16c^5 - 20c^3 + 5c + 1$. \checkmark)

Since $\cos(\pi/5) \neq -1$, we have $\cos(\pi/5)$ is a root of the
quartic factor:
\begin{equation}\label{eq:quartic}
  16c^4 - 16c^3 - 4c^2 + 4c + 1 = 0.
\end{equation}
This quartic factors further. The four roots of $\cos(5\theta) = -1$ with
$c \neq -1$ correspond to $\theta = \pi/5, 3\pi/5, 7\pi/5, 9\pi/5$,
i.e., $c = \cos(\pi/5), \cos(3\pi/5), \cos(7\pi/5), \cos(9\pi/5)$.
Since $\cos(7\pi/5) = \cos(2\pi - 7\pi/5) = \cos(3\pi/5)$ and
$\cos(9\pi/5) = \cos(2\pi - 9\pi/5) = \cos(\pi/5)$, there are only
two distinct values: $\cos(\pi/5)$ and $\cos(3\pi/5)$, each a double root.
The quartic therefore factors as:
\begin{equation}\label{eq:quartic-factor}
  16c^4 - 16c^3 - 4c^2 + 4c + 1
  = (4c^2 - 2c - 1)^2.
\end{equation}
(Verification: $(4c^2 - 2c - 1)^2 = 16c^4 - 16c^3 + 4c^2 - 8c^2 + 4c + 1
= 16c^4 - 16c^3 - 4c^2 + 4c + 1$. \checkmark)

Therefore $\cos(\pi/5)$ satisfies:
\begin{equation}\label{eq:quadratic-cos}
  4c^2 - 2c - 1 = 0.
\end{equation}
Multiplying by $4$: $16c^2 - 8c - 4 = 0$, or equivalently
$(4c)^2 - 2(4c) - 4 = 0$. With the substitution $u = 2c$, this is
$4u^2 - 4u - 4 = 0$, or $u^2 - u - 1 = 0$.

Alternatively, squaring $4c^2 - 2c - 1 = 0$ gives the original
relation~\eqref{eq:cheby-poly}: from $4c^2 = 2c + 1$, square both sides:
$16c^4 = 4c^2 + 4c + 1 = (2c + 1) + 4c + 1 = 6c + 2$,
wait---let us verify \eqref{eq:cheby-poly} directly.
The polynomial $16t^4 - 12t^2 + 1$ factors as
$(4t^2 - 2t - 1)(4t^2 + 2t - 1)$.
(Verification: $(4t^2 - 2t - 1)(4t^2 + 2t - 1) = 16t^4 + 8t^3 - 4t^2
- 8t^3 - 4t^2 + 2t - 4t^2 - 2t + 1 = 16t^4 - 12t^2 + 1$. \checkmark)

The roots of $4t^2 - 2t - 1 = 0$ are $t = (2 \pm \sqrt{4+16})/8
= (2 \pm \sqrt{20})/8 = (1 \pm \sqrt{5})/4$.
The roots of $4t^2 + 2t - 1 = 0$ are $t = (-2 \pm \sqrt{4+16})/8
= (-1 \pm \sqrt{5})/4$.

The four roots of $16t^4 - 12t^2 + 1 = 0$ are therefore:
\[
  \frac{1+\sqrt{5}}{4} = \frac{\varphi}{2}, \quad
  \frac{1-\sqrt{5}}{4} = \frac{\psi}{2}, \quad
  \frac{-1+\sqrt{5}}{4}, \quad
  \frac{-1-\sqrt{5}}{4}.
\]
These are $\cos(\pi/5)$, $\cos(3\pi/5)$, $\cos(7\pi/5)$, $\cos(9\pi/5)$
in some order.
\end{proof}

\begin{theorem}[$\cos(\pi/5) = \varphi/2$]\label{thm:cos-pi-5}
\begin{equation}\label{eq:cos-pi-5}
  \boxed{\; \cos\!\left(\frac{\pi}{5}\right) = \frac{\varphi}{2}
  = \frac{1+\sqrt{5}}{4} \;}
\end{equation}
\end{theorem}

\begin{proof}
From Lemma~\ref{lem:cheby}, $\cos(\pi/5)$ is a root of
$4t^2 - 2t - 1 = 0$ (the factor whose roots are
$(1 \pm \sqrt{5})/4$).
It remains to identify which root.

Since $\pi/5 = 36° \in (0°, 90°)$, we have $\cos(\pi/5) > 0$.
The two roots of $4t^2 - 2t - 1 = 0$ are:
\begin{itemize}[nosep]
  \item $(1 + \sqrt{5})/4 = \varphi/2 \approx 0.809 > 0$,
  \item $(1 - \sqrt{5})/4 = \psi/2 \approx -0.309 < 0$.
\end{itemize}
Therefore $\cos(\pi/5) = (1 + \sqrt{5})/4 = \varphi/2$.

We verify directly: $\cos 36° = \cos(\pi/5)$.
From the quadratic $4c^2 - 2c - 1 = 0$ with $c = \varphi/2$:
\[
  4\!\left(\frac{\varphi}{2}\right)^{\!2} - 2\!\left(\frac{\varphi}{2}\right) - 1
  = \varphi^2 - \varphi - 1
  = (\varphi + 1) - \varphi - 1
  = 0. \;\checkmark
\]
The third equality uses the axiom $\varphi^2 = \varphi + 1$.
\end{proof}

\begin{remark}[The Axiom in the Proof]\label{rem:axiom-link}
The identity $\cos(\pi/5) = \varphi/2$ holds \emph{because}
$\varphi$ satisfies $x^2 = x + 1$.
The verification reduces exactly to
$\varphi^2 - \varphi - 1 = 0$, which is the axiom.
No other property of $\varphi$ is used.
\end{remark}

\section{Pi from Phi}\label{sec:pi-from-phi}
% =============================================================================

\begin{theorem}[Pi as an Axiom Derivative]\label{thm:pi-exact}
\begin{equation}\label{eq:pi-exact}
  \boxed{\; \pi = p \cdot \arccos\!\left(\frac{\varphi}{\chi}\right)
  = 5\arccos\!\left(\frac{\varphi}{2}\right) \;}
\end{equation}
This is exact. No approximation, no series truncation, no numerical fit.
\end{theorem}

\begin{proof}
From Theorem~\ref{thm:cos-pi-5}:
\[
  \cos\!\left(\frac{\pi}{5}\right) = \frac{\varphi}{2} = \frac{\varphi}{\chi}.
\]
Since $\arccos$ is the inverse of $\cos$ on $[0, \pi]$, and
$\varphi/2 \in (0, 1) \subset [-1, 1]$:
\[
  \frac{\pi}{5} = \arccos\!\left(\frac{\varphi}{2}\right).
\]
Multiplying both sides by $5 = p$:
\[
  \pi = 5 \cdot \arccos\!\left(\frac{\varphi}{2}\right)
  = p \cdot \arccos\!\left(\frac{\varphi}{\chi}\right). \qedhere
\]
\end{proof}

\begin{corollary}[Non-Circularity]\label{cor:noncircular}
The identity $\pi = 5\arccos(\varphi/2)$ does not presuppose $\pi$.
\end{corollary}

\begin{proof}
The cosine function is defined independently of $\pi$, either as:
\begin{enumerate}[nosep]
  \item the power series $\cos(x) = \sum_{n=0}^{\infty} (-1)^n x^{2n}/(2n)!$, or
  \item the ratio of adjacent side to hypotenuse in a right triangle, or
  \item the unique solution of $y'' + y = 0$ with $y(0) = 1$, $y'(0) = 0$.
\end{enumerate}
In each definition, $\pi$ is the \emph{output}: it is the smallest
positive number $\omega$ such that $\cos(\omega) = -1$, or equivalently
$\cos(\omega/2) = 0$. Our identity recovers this number from
$\varphi$.

The axiom $x^2 = x + 1$ produces $\varphi$.
$\varphi$ produces $\varphi/2 = \varphi/\chi$.
The $\arccos$ function applied to $\varphi/\chi$ produces $\pi/5$.
Multiplication by $p = 5$ produces $\pi$.

The logical chain is:
\[
  x^2 = x + 1 \;\to\; \varphi \;\to\; \frac{\varphi}{\chi}
  \;\xrightarrow{\;\arccos\;}\; \frac{\pi}{p}
  \;\xrightarrow{\;\times\, p\;}\; \pi.
\]
Every constant ($\varphi$, $\chi$, $p$) derives from the axiom.
\end{proof}

\begin{corollary}[Pi Is Not Primitive]\label{cor:not-primitive}
$\pi$ is a derived constant. It is the angular content of the pentagonal
geometry that $x^2 = x + 1$ forces into existence: the regular pentagon
has interior angle $3\pi/5$ and central angle $2\pi/5$, both multiples
of $\pi/5 = \arccos(\varphi/2)$, where $\varphi$ and the pentagon are
consequences of the axiom.
\end{corollary}

\section{The Chudnovsky Base: $640320$ from the Fold Chain}
\label{sec:640320}
% =============================================================================

\begin{theorem}[Factorization of $640320$]\label{thm:640320}
\begin{equation}\label{eq:640320}
  \boxed{\;
  640320 = 2^{\chi d} \cdot d \cdot p \cdot 23 \cdot 29
  = 2^6 \cdot 3 \cdot 5 \cdot 23 \cdot 29
  \;}
\end{equation}
Every factor derives from $\{\chi, d, p\}$:
\begin{center}
\begin{tabular}{@{}cll@{}}
\toprule
Factor & Expression in $\{\chi, d, p\}$ & Value \\
\midrule
$2^6$ & $2^{\chi d} = 2^{2 \cdot 3}$ & $64$ \\
$3$ & $d = \chi^2 - 1$ & $3$ \\
$5$ & $p = \chi^2 + 1$ & $5$ \\
$23$ & $p^2 - \chi = 25 - 2$ & $23$ \\
$29$ & $\chi^2 + p^2 = 4 + 25$ & $29$ \\
\bottomrule
\end{tabular}
\end{center}
\end{theorem}

\begin{proof}
We verify the prime factorization of $640320$:
\begin{align*}
  640320 &= 64 \times 10005 \\
         &= 2^6 \times 10005 \\
         &= 2^6 \times 3 \times 3335 \\
         &= 2^6 \times 3 \times 5 \times 667 \\
         &= 2^6 \times 3 \times 5 \times 23 \times 29.
\end{align*}
Verification: $64 \times 3 = 192$; $192 \times 5 = 960$;
$960 \times 23 = 22080$; $22080 \times 29 = 640320$. \checkmark

The exponent of $2$ is $6 = \chi d = 2 \times 3$.
The remaining prime factors are $d = 3$, $p = 5$,
$p^2 - \chi = 23$, and $\chi^2 + p^2 = 29$.
Each is an arithmetic expression in $\chi$, $d$, $p$ alone.
\end{proof}

\begin{remark}[No External Constants]\label{rem:no-external}
The integer $640320$ contains no prime factor outside the
$\{\chi, d, p\}$-language. The primes $23$ and $29$ are not ad hoc:
$23$ is the backward fold $p^2 - \chi$ and $29$ is the terminal
value of the Pythagorean fold chain (Theorem~\ref{thm:fold-chain}).
\end{remark}

\begin{proposition}[Alternate Decomposition]\label{prop:640320-alt}
$640320 = 64 \times 10005$, where:
\begin{align}
  64 &= 2^{\chi d}, \label{eq:64} \\
  10005 &= d \cdot p \cdot 23 \cdot 29 = 3 \times 5 \times 23 \times 29.
  \label{eq:10005}
\end{align}
\end{proposition}

\begin{proof}
$3 \times 5 = 15$; $15 \times 23 = 345$; $345 \times 29 = 10005$.
Then $64 \times 10005 = 640320$. \checkmark
\end{proof}

\section{The Chudnovsky Formula}\label{sec:chudnovsky}
% =============================================================================

\begin{theorem}[Chudnovsky Series]\label{thm:chudnovsky}
The Chudnovsky brothers~\cite{chudnovsky1989} proved:
\begin{equation}\label{eq:chudnovsky}
  \frac{1}{\pi}
  = \frac{12}{C^{3/2}}
  \sum_{k=0}^{\infty}
  \frac{(-1)^k\, (6k)!\, (A + Bk)}{(3k)!\,(k!)^3\, C^{3k}}
\end{equation}
where
\begin{align}
  A &= 13591409, \label{eq:A} \\
  B &= 545140134, \label{eq:B} \\
  C &= 640320. \label{eq:C}
\end{align}
This series converges at a rate of approximately $14$ decimal digits per
term, the fastest known series for $1/\pi$.
\end{theorem}

\begin{proof}
The derivation is due to Chudnovsky and Chudnovsky~\cite{chudnovsky1989}.
It proceeds from the theory of complex multiplication applied to the
elliptic curve with $j$-invariant $j(\tau_{163}) = -640320^3$, where
$\tau_{163} = (1 + \sqrt{-163})/2$ and $163$ is the largest Heegner
number. The modular parametrization yields the hypergeometric structure
$(6k)!/((3k)!(k!)^3)$ and determines the constants $A$, $B$, $C$ from
the singular value of the $j$-function. See Borwein and
Borwein~\cite{borwein1987} for the complete class field theory
derivation.
\end{proof}

\subsection{The Constants $A$ and $B$}

\begin{theorem}[Factorization of $B$]\label{thm:B-factor}
The linear coefficient factors as:
\begin{equation}\label{eq:B-factor}
  B = 545140134 = 2 \cdot 3^2 \cdot 7 \cdot 11 \cdot 19 \cdot 127
      \cdot 163.
\end{equation}
\end{theorem}

\begin{proof}
Direct factorization:
\begin{align*}
  545140134 &= 2 \times 272570067, \\
  272570067 &= 3 \times 90856689, \\
  90856689 &= 3 \times 30285563, \\
  30285563 &= 7 \times 4326509, \\
  4326509 &= 11 \times 393319, \\
  393319 &= 19 \times 20701, \\
  20701 &= 127 \times 163.
\end{align*}
Verification: $2 \times 9 \times 7 \times 11 \times 19 \times 127 \times
163 = 545140134$. \checkmark
\end{proof}

\begin{theorem}[Dodecahedral Interpretation of $B$]\label{thm:B-dodec}
Every prime factor of $B$ is a named invariant of the dodecahedron $\{5, 3\}$:
\begin{center}
\begin{tabular}{@{}cll@{}}
\toprule
Factor & Dodecahedral identity & Value \\
\midrule
$2$ & $\chi = V - E + F$ & Euler characteristic \\
$3$ & $d$ (vertex degree) & Schl\"afli $q$ \\
$7$ & $L_4 = \varphi^4 + \psi^4 = V - F - 1$ & Fourth Lucas number \\
$11$ & $b_1 = E - V + 1 = L_5$ & First Betti number \\
$19$ & $V - 1 = 20 - 1$ & Nontrivial vertex count \\
$127$ & $2^{L_4} - 1 = 2^7 - 1$ & Mersenne prime from $L_4$ \\
$163$ & Largest Heegner number & Class number one discriminant \\
\bottomrule
\end{tabular}
\end{center}
\end{theorem}

\begin{proof}
We verify each identification.

\textbf{$\boldsymbol{L_4 = 7}$:} The Lucas numbers satisfy $L_n = \varphi^n + \psi^n$.
Then $L_4 = \varphi^4 + \psi^4$.
Since $\varphi^2 = \varphi + 1$ and $\psi^2 = \psi + 1$,
we get $\varphi^4 = (\varphi + 1)^2 = \varphi^2 + 2\varphi + 1 = 3\varphi + 2$
and $\psi^4 = 3\psi + 2$.
So $L_4 = 3(\varphi + \psi) + 4 = 3(1) + 4 = 7$.
Also $V - F - 1 = 20 - 12 - 1 = 7$. \checkmark

\textbf{$\boldsymbol{b_1 = 11}$:} The first Betti number of the dodecahedral graph is
$b_1 = E - V + 1 = 30 - 20 + 1 = 11$.
Also $L_5 = \varphi^5 + \psi^5 = \varphi \cdot \varphi^4 + \psi \cdot \psi^4
= \varphi(3\varphi + 2) + \psi(3\psi + 2)
= 3(\varphi^2 + \psi^2) + 2(\varphi + \psi)
= 3 \cdot 3 + 2 \cdot 1 = 11$.
(Here $\varphi^2 + \psi^2 = (\varphi + \psi)^2 - 2\varphi\psi = 1 + 2 = 3$.) \checkmark

\textbf{$\boldsymbol{V - 1 = 19}$:} $20 - 1 = 19$. \checkmark

\textbf{$\boldsymbol{2^7 - 1 = 127}$:} $128 - 1 = 127$. Primality: $127$ is not
divisible by $2, 3, 5, 7, 11$ (and $\lfloor\sqrt{127}\rfloor = 11$). \checkmark

\textbf{$\boldsymbol{163}$:} The Heegner numbers are those $D > 0$ for which
$\mathbb{Q}(\sqrt{-D})$ has class number one. By the
Stark--Heegner theorem~\cite{stark1967},
there are exactly nine: $1, 2, 3, 7, 11, 19, 43, 67, 163$.
The value $163$ is the largest, and it determines the CM point
$\tau_{163} = (1 + \sqrt{-163})/2$ at which
$j(\tau_{163}) = -640320^3$~\cite{chudnovsky1989}.
\end{proof}

\end{equation}

From the $q$-expansion of the $j$-function at the CM point $\tau_{163}$,

in the precise sense of the Borwein--Borwein
theory~\cite{borwein1987}. The value $13591409$ is an algebraic integer

Since $\mathbb{Q}(\sqrt{-163})$ has class number one, this field is
$\mathbb{Q}(\sqrt{-163})$ itself, and $A$ is rational---in fact, an integer.
\end{theorem}

\begin{proof}
The computation is carried out in full in~\cite{chudnovsky1989}

\end{proof}

\section{The Binary Icosahedral Coefficient}\label{sec:2I}
% =============================================================================

\begin{theorem}[The $k = 1$ Coefficient]\label{thm:120}
The multinomial coefficient at $k = 1$ in the Chudnovsky series is:
\begin{equation}\label{eq:120}
  \boxed{\;
  \frac{(6 \cdot 1)!}{(3 \cdot 1)!\,(1!)^3}
  = \frac{6!}{3!}
  = \frac{720}{6}
  = 120
  = |2I|
  \;}
\end{equation}
where $2I$ is the binary icosahedral group, the universal central
extension of $A_5$ (the rotation group of the dodecahedron), a subgroup
of $\mathrm{SU}(2)$ of order $120$.
\end{theorem}

\begin{proof}
The computation: $6! = 720$, $3! = 6$, $1!^3 = 1$,
$720/6 = 120$.

The group order: the rotation group of the dodecahedron $\{5,3\}$ is
$A_5$ with $|A_5| = 60 = V \cdot d = F \cdot p = 20 \times 3 = 12 \times 5$.
The binary icosahedral group $2I$ is the preimage of $A_5$ under the
double cover $\mathrm{SU}(2) \to \mathrm{SO}(3)$, so
$|2I| = 2|A_5| = 120$.
\end{proof}

\begin{remark}[Structural, Not Coincidental]\label{rem:structural}
The Chudnovsky series arises from Klein's icosahedral function, which
encodes the invariant theory of $2I$ acting on $\mathbb{C}^2$.
Klein~\cite{klein1884} showed that $2I$ produces polynomial invariants
of degrees $F = 12$, $V = 20$, $E = 30$ (the face, vertex, and edge
counts of the dodecahedron). The modular $j$-function is built from
these invariants. The appearance of $|2I| = 120$ as the first
nontrivial coefficient of the Chudnovsky series---a series derived from
the $j$-function---is a consequence of this structural chain, not a
numerical coincidence.
\end{remark}

  \item \textbf{Fold Primes $\to$ $C$:}

  \item \textbf{$C \to 1/\pi$:} The Chudnovsky series with $C = 640320$

\end{enumerate}

The direct path~\eqref{eq:direct}:
\begin{enumerate}[nosep]
  \item \textbf{Axiom $\to \varphi$:} As above.
  \item \textbf{$\varphi \to \pi$:}
    $\cos(\pi/5) = \varphi/2$ (Theorem~\ref{thm:cos-pi-5}),

The spectral theory of the dodecahedron provides independent confirmation

\begin{definition}[Dodecahedral Laplacian]\label{def:laplacian}
Let $A$ be the $20 \times 20$ adjacency matrix of the dodecahedral graph,
and $L = dI - A = 3I - A$ the combinatorial Laplacian. Its eigenvalues
are:
\begin{center}
\begin{tabular}{@{}ccc@{}}
\toprule
Eigenvalue $\lambda_i$ & Multiplicity & \\
\midrule
$0$ & $1$ & Trivial (connected graph) \\
$3 - \sqrt{5}$ & $3$ & Golden pair (lower) \\
$2$ & $5$ & \\
$3$ & $4$ & Vertex degree \\
$5$ & $4$ & Face sides \\
$3 + \sqrt{5}$ & $3$ & Golden pair (upper) \\
\bottomrule
\end{tabular}
\end{center}
Total multiplicity: $1 + 3 + 5 + 4 + 4 + 3 = 20 = V$.
\end{definition}

\begin{theorem}[Trace Identity]\label{thm:trace}
\begin{equation}\label{eq:trace-137}
  dp \cdot \Tr(L^+) = 15 \times \frac{137}{15} = 137
\end{equation}
where $L^+$ is the Moore--Penrose pseudoinverse of $L$.
\end{theorem}

\begin{proof}
$L^+$ inverts each nonzero eigenvalue:
\[
  \Tr(L^+)
  = \frac{3}{3-\sqrt{5}} + \frac{5}{2} + \frac{4}{3}
    + \frac{4}{5} + \frac{3}{3+\sqrt{5}}.
\]
The golden pair contributes:
\[
  \frac{3}{3-\sqrt{5}} + \frac{3}{3+\sqrt{5}}
  = \frac{3(3+\sqrt{5}) + 3(3-\sqrt{5})}{(3-\sqrt{5})(3+\sqrt{5})}
  = \frac{18}{9-5} = \frac{18}{4} = \frac{9}{2}.
\]
Summing all terms with common denominator $30$:
\[
  \Tr(L^+) = \frac{9}{2} + \frac{5}{2} + \frac{4}{3} + \frac{4}{5}
  = \frac{135 + 75 + 40 + 24}{30}
  = \frac{274}{30}
  = \frac{137}{15}.
\]
Then $dp \cdot \Tr(L^+) = 15 \cdot 137/15 = 137$.
\end{proof}

\begin{theorem}[Heat Kernel on the Poincar\'e Dodecahedral Space]
\label{thm:heat}
The Poincar\'e dodecahedral space $M = S^3/2I$ (the quotient of the
$3$-sphere by the binary icosahedral group acting freely) has heat kernel
trace
\begin{equation}\label{eq:heat}
  K(t) := \Tr\!\left(e^{-t\Delta_M}\right)
  = \frac{\sqrt{\pi}}{480\, t^{3/2}} + O(t^{-1/2})
  \quad\text{as } t \to 0^+
\end{equation}
where $480 = 4 \cdot |2I| = 4 \times 120$.
\end{theorem}

\begin{proof}
Weyl's asymptotic law: on a compact $n$-manifold $M$,
$K(t) \sim \mathrm{vol}(M)/(4\pi t)^{n/2}$ as $t \to 0^+$.
With $n = 3$:
\[
  \mathrm{vol}(S^3/2I) = \frac{\mathrm{vol}(S^3)}{|2I|}
  = \frac{2\pi^2}{120} = \frac{\pi^2}{60}.
\]
Therefore:
\[
  K(t) \sim \frac{\pi^2/60}{(4\pi t)^{3/2}}
  = \frac{\pi^2}{60 \cdot 8\pi^{3/2} t^{3/2}}
  = \frac{\sqrt{\pi}}{480\, t^{3/2}}. \qedhere
\]
\end{proof}

\section{The Arctangent Decomposition}\label{sec:arctan}
% =============================================================================

\begin{proposition}[Pi from $1$, $\chi$, $d$]\label{prop:arctan}
\begin{equation}\label{eq:arctan}
  \pi = \arctan(1) + \arctan(\chi) + \arctan(d)
  = \arctan(1) + \arctan(2) + \arctan(3).
\end{equation}
\end{proposition}

\begin{proof}
Let $\alpha = \arctan(1)$, $\beta = \arctan(2)$, $\gamma = \arctan(3)$.
By the tangent addition formula:
\[
  \tan(\alpha + \beta)
  = \frac{\tan\alpha + \tan\beta}{1 - \tan\alpha \cdot \tan\beta}
  = \frac{1 + 2}{1 - 1 \cdot 2}
  = \frac{3}{-1} = -3.
\]
Since $\alpha + \beta \in (\pi/4 + \arctan 2) \subset (\pi/2, \pi)$
and $\tan(\alpha + \beta) = -3 < 0$, we have
$\alpha + \beta = \pi - \arctan 3 = \pi - \gamma$.
Therefore $\alpha + \beta + \gamma = \pi$.
\end{proof}

\begin{remark}\label{rem:arctan-framework}
The arguments $1$, $2$, $3$ are respectively: the unit, $\chi$, and $d$.
All three derive from the axiom. This gives a second exact, closed-form
expression for $\pi$ in the $\{\chi, d, p\}$-language.
\end{remark}

\section{Conclusion}\label{sec:conclusion}
% =============================================================================

The axiom $x^2 = x + 1$ determines $\pi$ through two independent,
completed derivations:

\begin{enumerate}
  \item \textbf{The Direct Path (Exact, Algebraic):}
    \[
      x^2 = x + 1 \;\to\; \varphi \;\to\; \cos(\pi/5) = \varphi/2
      \;\to\; \pi = 5\arccos(\varphi/2).
    \]
    The proof that $\cos(\pi/5) = \varphi/2$ reduces to the axiom
    $\varphi^2 - \varphi - 1 = 0$ (Theorem~\ref{thm:cos-pi-5}).

  \item \textbf{The Arithmetic Path (Chudnovsky):}
    \[
      x^2 = x + 1 \;\to\; \chi, d, p \;\to\; 23, 29
      \;\to\; 640320 = 2^{\chi d} \cdot d \cdot p \cdot 23 \cdot 29
      \;\to\; \frac{1}{\pi}.
    \]
    The base $C = 640320$ of the fastest known series for $1/\pi$
    factors entirely through the $\{\chi, d, p\}$-language
    (Theorem~\ref{thm:640320}).
\end{enumerate}

A third confirmation comes from the spectral theory of $S^3/2I$
(Section~\ref{sec:spectral}), where the heat kernel trace produces
$\sqrt{\pi}$ from dodecahedral data alone.

The first nontrivial Chudnovsky coefficient is
$120 = |2I|$ (Theorem~\ref{thm:120}), the order of the binary
icosahedral group---the double cover of the rotation group of the
dodecahedron that the axiom forces. The linear coefficient $B = 545140134$
factors entirely through dodecahedral primes
(Theorem~\ref{thm:B-dodec}), and $L_4 = 7$ divides $B$ in every known
level-6 Ramanujan--Sato series (Proposition~\ref{prop:seven}).

$\pi$ is not an independent constant. It is the angular content of the
pentagonal geometry that $x^2 = x + 1$ forces into existence.
