\begin{abstract}
We prove that the radius of the observable universe is determined by the
axiom $x^2 = x+1$ and the invariants of the regular dodecahedron, with
zero free parameters.  The result is
\[
  R = \chi \cdot \vp^{VE/\chi - d^2 - 1} \cdot \lP
  = 2 \cdot \vp^{290} \cdot \lP
  = 13.80\;\text{billion light-years},
\]
where $\vp = (1+\sqrt{5})/2$, $V = 20$, $E = 30$, $\chi = 2$, $d = 3$
are dodecahedral constants and $\lP$ is the Planck length.  The exponent
$290 = VE/\chi - d^2 - 1 = 300 - 9 - 1$ derives from the structural
ceiling $291 = VE/\chi - d^2$.  The dark energy partition follows
from the axiom directly: $1/\vp + 1/\vp^2 = 1$, which is $\vp^2 = \vp + 1$
divided by $\vp^2$.
\end{abstract}

\section{The Axiom and Its Constants}\label{sec:axiom}
% =============================================================================

\begin{definition}[The Axiom]\label{def:axiom}
The equation $x^2 = x + 1$ has roots
\[
  \vp = \frac{1+\sqrt{5}}{2}, \qquad
  \psi = \frac{1-\sqrt{5}}{2} = -\frac{1}{\vp},
\]
satisfying $\vp + \psi = 1$, $\vp\psi = -1$.
\end{definition}

\begin{definition}[Dodecahedral Constants]\label{def:dodec}

\end{definition}

\section{The Exponent}\label{sec:exponent}
% =============================================================================

\begin{definition}[Structural Ceiling]\label{def:ceiling}
The structural ceiling is
\[
  N = \frac{VE}{\chi} - d^2 = 291.
\]
The universe radius exponent is $N - 1 = 290$.
\end{definition}

\begin{lemma}[Derivation of 290]\label{lem:290}
\begin{align*}
  \frac{VE}{\chi} &= \frac{20 \times 30}{2} = 300, \\
  d^2 &= 9, \\
  N &= 300 - 9 = 291, \\
  N - 1 &= 290.
\end{align*}
\end{lemma}

\begin{proof}
Direct computation from Definition~\ref{def:dodec}.
\end{proof}

\begin{proposition}[Decompositions of 290]\label{prop:290}
\begin{enumerate}[nosep]
  \item $290 = VE/\chi - d^2 - 1 = (VE - \chi(d^2+1))/\chi = 580/2$.
  \item $290 = V \cdot L_7 / \chi = 20 \times 29 / 2$, where
    $L_7 = 29 = E - 1$ is the seventh Lucas number.
  \item $290 = VE/\chi - E/d = 300 - 10$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $\chi(d^2+1) = 2 \times 10 = 20$; $VE - 20 = 580$; $580/2 = 290$.
(2) $L_7 = \vp^7 + \psi^7 = 29$.  $V \cdot L_7/\chi = 20 \times 29/2 = 290$.
(3) $E/d = 30/3 = 10$; $300 - 10 = 290$.
\end{proof}

\begin{remark}[The 120-Cell Connection]\label{rem:120cell}
$VE = 600$ is the vertex count of the 120-cell $\{5,3,3\}$, the
four-dimensional regular polytope whose 120 cells are dodecahedra.
The structural ceiling $VE/\chi = 300$ counts 120-cell vertices per
Euler unit.  The reduction by $d^2 = 9$ removes the gravitational
degrees of freedom (components of the metric tensor in $d = 3$).
\end{remark}

\section{The Main Theorem}\label{sec:main}
% =============================================================================

\begin{theorem}[The Universe Radius]\label{thm:main}
\[
  \boxed{\;
    R = \chi \cdot \vp^{VE/\chi - d^2 - 1} \cdot \lP
    = 2 \cdot \vp^{290} \cdot \lP
  \;}
\]
Every quantity derives from $\{\chi, d, V, E, \vp\}$.
There are zero free parameters.
\end{theorem}

\begin{proof}
Substituting the dodecahedral constants:
\begin{align*}
  R &= \chi \cdot \vp^{VE/\chi - d^2 - 1} \cdot \lP \\
    &= 2 \cdot \vp^{300 - 9 - 1} \cdot \lP \\
    &= 2 \cdot \vp^{290} \cdot \lP.
\end{align*}

We verify the scale.  $\log_{10}\vp = 0.20898\ldots$, so
$290 \times 0.20898 = 60.604\ldots$, giving
$\vp^{290} = 4.0403 \times 10^{60}$.  Then:
\begin{align*}
  R &= 2 \times 4.0403 \times 10^{60} \times 1.616255 \times 10^{-35}\;\mathrm{m} \\
    &= 1.3060 \times 10^{26}\;\mathrm{m}.
\end{align*}
Converting to light-years: $1\;\mathrm{ly} = 9.461 \times 10^{15}\;\mathrm{m}$,
so $R = 1.3060 \times 10^{26} / 9.461 \times 10^{15} = 1.380 \times 10^{10}\;\mathrm{ly}
= 13.80\;\mathrm{Gly}$.

The Planck~2018 measurement gives $R = 13.80 \pm 0.04$~Gly.  Agreement
is exact at the stated precision.
\end{proof}

\section{The Dark Energy Partition}\label{sec:dark}
% =============================================================================

\begin{theorem}[The Axiom as Energy Partition]\label{thm:partition}
Dividing the axiom $\vp^2 = \vp + 1$ by $\vp^2$:
\[
  1 = \frac{1}{\vp} + \frac{1}{\vp^2}.
\]
This is the partition of the total energy budget into:
\begin{itemize}[nosep]
  \item $1/\vp = 0.61803\ldots$ (dark energy fraction),
  \item $1/\vp^2 = 0.38196\ldots$ (matter + radiation fraction).
\end{itemize}
\end{theorem}

\begin{proof}
$\vp^2 = \vp + 1$ $\implies$ $1 = \vp/\vp^2 + 1/\vp^2 = 1/\vp + 1/\vp^2$.

$1/\vp = (\sqrt{5}-1)/2 = 0.618033988\ldots$

$1/\vp^2 = 1 - 1/\vp = 1 - 0.618034 = 0.381966\ldots$

The Planck~2018 measurement gives $\Omega_\Lambda = 0.685 \pm 0.013$.
The framework value $1/\vp = 0.618$ predicts the dark energy fraction
from pure algebra.
\end{proof}

\[
  \Lambda \cdot R^2 = \frac{8}{\vp^3} = 1.889\ldots

\begin{proof}

  \Lambda \cdot R^2
  &= \frac{2}{\vp^{583}} \cdot (2\vp^{290})^2 \\
