\begin{abstract}
Starting from the Pythagorean axiom $a^2 + b^2 = c^2$ and nothing else,
we derive the radius of the observable universe with zero free parameters.
The axiom forces the golden ratio $\vp = (1+\sqrt{5})/2$ as its
self-similar fixed point, forces the regular dodecahedron as the unique
polyhedron built from $\vp$, and forces the exponent $290$ from
dodecahedral invariants.  The Planck length $\lP$ enters as the minimum
state change (the number $2$ in framework units---the smallest Pythagorean
move).  The result:
\[
  R \;=\; 2 \cdot \vp^{290} \cdot \lP
  \;=\; 1.306 \times 10^{26}\;\text{m}
  \;=\; 13.80\;\text{Gly}.
\]
The Planck~2018 measurement gives $13.787 \pm 0.020$~Gly.  The
theoretical value lies within $0.9\sigma$ of the observed central
value.  Every quantity in the formula traces back to the axiom through
an explicit chain of derivations.
\end{abstract}

\section{The Axiom}\label{sec:axiom}
% =============================================================================

\begin{axiom}[The Pythagorean Theorem]\label{ax:pyth}
For a right triangle with legs $a, b$ and hypotenuse $c$:
\[
  a^2 + b^2 = c^2.
\]
This is the starting point.  Every result below is derived from this
statement.  No external formula is imported.
\end{axiom}

\begin{remark}[What the axiom says]
The axiom equates the \emph{squared} measure (the normalizer: sign-free,
always positive) with the \emph{sum of parts}.  Squaring and adding are
the two primitive operations.  The axiom says they balance.
\end{remark}

\section{From the Axiom to the Golden Ratio}\label{sec:golden}
% =============================================================================

\begin{definition}[Self-Similar Triangle]\label{def:selfsim}
A right triangle is \emph{self-similar under the axiom} if removing one
leg from the hypotenuse produces a triangle similar to the original.
That is, the triangle $(a, b, c)$ satisfies both
\[
  a^2 + b^2 = c^2 \qquad\text{and}\qquad \frac{c}{a} = \frac{a}{c - a}.
\]
\end{definition}

\begin{lemma}[The Golden Ratio from Self-Similarity]\label{lem:golden}
If $x = c/a$ is the ratio of hypotenuse to leg in a self-similar right
triangle, then $x$ satisfies $x^2 = x + 1$.
\end{lemma}

\begin{proof}
From the self-similarity condition $c/a = a/(c-a)$, cross-multiply:
\[
  c(c - a) = a^2, \quad\text{i.e.,}\quad c^2 - ac = a^2.
\]
Divide both sides by $a^2$ and set $x = c/a$:
\[
  x^2 - x = 1, \qquad\text{i.e.,}\qquad x^2 = x + 1.
\]
The positive root is $\vp = (1+\sqrt{5})/2 = 1.6180339887\ldots$

The negative root is $\psi = (1-\sqrt{5})/2 = -1/\vp$.

The sum and product of the roots:
\[
  \vp + \psi = 1, \qquad \vp\psi = -1.
\]
Both follow from Vi\`{e}ta's formulas applied to $x^2 - x - 1 = 0$.
\end{proof}

\begin{remark}
The equation $x^2 = x + 1$ is the Pythagorean axiom specialized to
self-similarity.  It says: \emph{the square of the whole equals the
whole plus the unit}.  Squaring $=$ adding.  This is the axiom restated
for the ratio that preserves itself.
\end{remark}

\begin{proposition}[The Dodecahedron Is Forced]\label{prop:dodec}
The regular dodecahedron is the unique Platonic solid whose edge
lengths, face diagonals, and dihedral angles are all expressible as
algebraic functions of $\vp$ alone.
\end{proposition}

\begin{proof}

\begin{center}

Solid & Edge/Circumradius involves \\
\midrule

Cube & $2/\sqrt{3}$ \\

Dodecahedron & $4/(\vp^2\sqrt{3}) = 4/((1+\vp)\sqrt{3})$ \\
Icosahedron & $2/\vp$ (but dual to dodecahedron) \\
\bottomrule
\end{tabular}
\end{center}

The dodecahedron's face is a regular pentagon.  The diagonal-to-side
ratio of a regular pentagon is exactly $\vp$ (this is the geometric
definition of the golden ratio, itself derivable from the axiom via
Lemma~\ref{lem:golden}).  The dodecahedron has pentagonal faces;
the icosahedron has pentagonal vertex figures.  Both are built from
$\vp$.  The dodecahedron is the face-primary solid (the golden ratio
appears in its \emph{faces}, not just its vertices), so we take the
dodecahedron as the canonical $\vp$-solid.

Its invariants:
\begin{center}
\begin{tabular}{@{}lll@{}}

$p$ & $5$ & Sides per face (pentagon) \\
$d$ & $3$ & Edges per vertex (valency = spatial dimension) \\
$\chi$ & $2$ & Euler characteristic: $V - E + F = 20 - 30 + 12$ \\
\bottomrule
\end{tabular}
\end{center}
\end{proof}

\begin{remark}
These six values $\{V, E, F, p, d, \chi\}$ are not free parameters.
They are forced by the requirement that $\vp$ tiles a convex solid.
The Schl\"{a}fli symbol $\{p, d\} = \{5, 3\}$ encodes that each face
is a pentagon ($p = 5$) and each vertex meets $d = 3$ faces.  The
remaining counts follow from Euler's formula.
\end{remark}

\section{The Pythagorean Reading of the Dodecahedron}\label{sec:pyth-dodec}

\begin{lemma}[Squared and Linear Readings]\label{lem:readings}
The roots $\vp$ and $\psi$ provide two readings of any power:
\begin{align}
  \vp^n + \psi^n &= L_n \quad\text{(the $n$th Lucas number --- the Pythagorean sum)}, \label{eq:lucas}\\
  \vp^n \cdot \psi^n &= (-1)^n \quad\text{(the product --- the normalization)}. \label{eq:product}
\end{align}
\end{lemma}

\begin{proof}
Both follow from $\vp + \psi = 1$ and $\vp\psi = -1$ by induction
and the recurrence $L_{n+1} = L_n + L_{n-1}$, which itself is
$x^2 = x + 1$ applied to the index.
\end{proof}

\begin{corollary}[Key Lucas Values from the Dodecahedron]\label{cor:lucas}
\begin{align*}
  L_1 &= 1, \quad L_2 = 3 = d, \quad L_3 = 4 = \chi^2, \quad
  L_4 = 7 = E - 23, \\
  L_5 &= 11, \quad L_6 = 18, \quad L_7 = 29 = E - 1.
\end{align*}
\end{corollary}

\section{Deriving the Exponent 290}\label{sec:exponent}
% =============================================================================

\begin{definition}[The Structural Ceiling]\label{def:ceiling}
Define the \emph{structural ceiling} of the dodecahedron:
\[
  N = \frac{VE}{\chi} - d^2.
\]
\end{definition}

\begin{lemma}[Computation of $N$]\label{lem:291}
\begin{align*}
  VE &= 20 \times 30 = 600, \\
  \frac{VE}{\chi} &= \frac{600}{2} = 300, \\
  d^2 &= 3^2 = 9, \\
  N &= 300 - 9 = 291.
\end{align*}
\end{lemma}

\begin{remark}[Why each term]\label{rem:terms}
\begin{enumerate}[nosep]
  \item $VE = 600$ is the vertex count of the 120-cell $\{5,3,3\}$,
    the four-dimensional polytope whose 120 cells are dodecahedra.
    This is the total combinatorial capacity of the $\vp$-lattice
    extended one dimension.
  \item Division by $\chi = 2$ normalizes by the Euler parity.
    The Pythagorean axiom has two legs; the Euler characteristic
    counts the two-sidedness of any closed surface.  Dividing by
    $\chi$ converts the raw capacity to single-sided measure.
  \item Subtraction of $d^2 = 9$ removes the gravitational degrees
    of freedom.  In $d = 3$ spatial dimensions, the symmetric metric
    tensor has $d(d+1)/2 = 6$ independent components, and the
    Riemann curvature adds $d(d-1)/2 = 3$ more constraints.
    The full count is $d^2 = 9$.  These are internal---they do not
    contribute to the \emph{extent} of the universe but to its
    \emph{local curvature}.
\end{enumerate}
\end{remark}

\begin{proposition}[Why 290]\label{prop:290}
The universe radius exponent is
\[
  N - 1 = 291 - 1 = 290.
\]
The $-1$ is the axiom seed.  The ceiling $N = 291$ counts the total
steps from the axiom ($+1$, the seed) through $290$ propagation steps.
The radius is the propagation, not the total: the universe is $290$
golden powers wide, launched from the seed $+1$.

Equivalently:
\[
  290 = \frac{VE}{\chi} - d^2 - 1 = \frac{VE - \chi(d^2 + 1)}{\chi}
  = \frac{600 - 20}{2} = \frac{580}{2}.
\]
\end{proposition}

\begin{proposition}[Arithmetic Structure of 290]\label{prop:factor}
\[
  290 = 2 \times 145 = 2 \times 5 \times 29.
\]
In dodecahedral language:
\begin{enumerate}[nosep]
  \item $290 = \chi \times p \times L_7$, where $\chi = 2$, $p = 5$,
    $L_7 = 29$.
  \item The factor $\chi = 2$: the Pythagorean axiom has two legs.
    The universe has two conjugate directions (the $\vp$ leg and the
    $\psi$ leg; equivalently, space and time, or matter and radiation).
  \item The factor $p = 5$: the pentagon, the face of the dodecahedron,
    the geometric incarnation of $\vp$.
  \item The factor $L_7 = 29$: the seventh Lucas number.  Reached
    by three cube-depth jumps from $L_1 = 1$:
    $L_1 = 1 \to L_4 = 7 \to L_7 = 29$.  Each jump traverses $d = 3$
    steps of the recurrence.  Three jumps $= d$ jumps of size $d$.
    The Lucas chain reaches $29$ by the cube acting on itself.
  \item $291 = \chi \cdot p \cdot L_7 + 1$: the ceiling is the
    propagation ($290$) plus the axiom seed ($+1$).
\end{enumerate}
\end{proposition}

  \item \textbf{Algebraic:} From $\vp^3 + \psi^3 = L_3 = 4 = \chi^2$
    and $\vp^3 \cdot \psi^3 = -1$, the sum of cubes gives
    $\chi + 2 = \chi^2$, forcing $\chi = 2$.
\end{enumerate}
\end{proposition}

\begin{theorem}[The Universe Radius]\label{thm:main}
\[
  \boxed{\;
    R = \chi \cdot \vp^{\,VE/\chi \;-\; d^2 \;-\; 1} \cdot \lP
    = 2 \cdot \vp^{290} \cdot \lP
  \;}
\]
Every quantity derives from the Pythagorean axiom:
\begin{itemize}[nosep]
  \item $\vp$: forced by self-similarity under $a^2 + b^2 = c^2$
    (Lemma~\ref{lem:golden}).
  \item $V, E, F, d, \chi$: forced by requiring $\vp$ to tile a
    convex solid (Proposition~\ref{prop:dodec}).
  \item The exponent $290$: forced by dodecahedral arithmetic

  \item The prefactor $2$: forced by the two-legged structure of the
    axiom (Proposition~\ref{prop:chi}).
  \item $\lP$: the minimum state change (Proposition~\ref{prop:planck}).
\end{itemize}
There are zero free parameters.
\end{theorem}

\begin{proof}
The structure of the proof is:
\[
  \text{Axiom} \;\xrightarrow{\text{self-sim.}}\; \vp
  \;\xrightarrow{\text{tiling}}\; \text{dodecahedron}
  \;\xrightarrow{\text{arithmetic}}\; 290
  \;\xrightarrow{\text{assembly}}\; R.

Substituting the dodecahedral constants:
\begin{align*}
  R &= \chi \cdot \vp^{VE/\chi - d^2 - 1} \cdot \lP \\

    &= 2 \cdot \vp^{300 - 9 - 1} \cdot \lP \\
    &= 2 \cdot \vp^{290} \cdot \lP. \qedhere
\end{align*}
\end{proof}

\begin{theorem}[The Axiom as Energy Partition]\label{thm:partition}
Divide the axiom $\vp^2 = \vp + 1$ by $\vp^2$:
\[
  1 = \frac{1}{\vp} + \frac{1}{\vp^2}.
\]
This partitions unity into two fractions:
\begin{itemize}[nosep]
  \item $1/\vp = 0.61803\ldots$ \;---\; the dark energy fraction,
  \item $1/\vp^2 = 0.38197\ldots$ \;---\; the matter $+$ radiation fraction.
\end{itemize}
\end{theorem}

\begin{proof}
Starting from the axiom:
\[
  \vp^2 = \vp + 1 \quad\implies\quad
  \frac{\vp^2}{\vp^2} = \frac{\vp}{\vp^2} + \frac{1}{\vp^2}
  \quad\implies\quad
  1 = \frac{1}{\vp} + \frac{1}{\vp^2}.

Numerically:
\[
  \frac{1}{\vp} = \frac{\sqrt{5}-1}{2} = 0.6180339887\ldots, \qquad
  \frac{1}{\vp^2} = \frac{3-\sqrt{5}}{2} = 0.3819660112\ldots

Sum: $0.6180\ldots + 0.3820\ldots = 1.0000\ldots$ \;(exact by the axiom).
\end{proof}

\begin{remark}
