% -- Abstract ----------------------------------------------------------------
\begin{abstract}
We establish the complete algebraic structure forced by the axiom
$x^{2} = x + 1$.  Starting from this single equation, we construct
the number field $\Q(\sqrt{5})$, the ring of integers $\Z[\vp]$,
the Galois automorphism $\sigma : \vp \mapsto \ps$, the
exponential embedding $x \mapsto \vp^{x}$, and the Fibonacci
recurrence as the canonical dynamics on $\Z[\vp]$.  Every theorem
is proved completely.  We establish:
(1)~$\Z[\vp]$ is a Euclidean domain, hence a PID and UFD;
(2)~the norm $\Norm(a + b\vp) = a^{2} + ab - b^{2}$ is multiplicative
and always integer;
(3)~the Galois involution $\sigma$ fixes exactly $\Q$ and the
point $1/2 = (\vp + \ps)/2$;
(4)~the map $x \mapsto \vp^{x}$ is a homomorphism from $(\R, +)$
to $(\R_{>0}, \times)$;
(5)~inversion is exact: $\vp^{-n} = |\ps|^{n}$ for all $n \ge 0$;
(6)~the Fibonacci recurrence over $\F_{2}$ has period $3 = d$
(the Pisano period $\pi(2)$);
(7)~the dodecahedron $\{5,3\}$ is the unique Platonic solid whose
geometry is built from $\vp$, and its rotation group $A_{5}$ is
the quintic barrier.
The machine state $(x, y) = (n, n - 2k)$ with embedding $y_{k} = \vp^{x_{k}}$
terminates the multiplicative dynamics at step $291$, where
$\ps^{291} < \ell_{\mathrm{P}}$ (the Planck length).
\end{abstract}

\section{The Axiom}\label{sec:axiom}
% =============================================================================

\begin{definition}[The Axiom]\label{def:axiom}
The foundational equation is
\begin{equation}\label{eq:axiom}
  x^{2} = x + 1,
\end{equation}
equivalently $x^{2} - x - 1 = 0$, with discriminant
$\Delta = 1 + 4 = 5$ and roots
\begin{equation}\label{eq:roots}
  \vp = \frac{1 + \sqrt{5}}{2} = 1.618033988\ldots, \qquad
  \ps = \frac{1 - \sqrt{5}}{2} = -0.618033988\ldots
\end{equation}
\end{definition}

\begin{lemma}[Vieta Relations]\label{lem:vieta}
$\vp + \ps = 1$ and $\vp \cdot \ps = -1$.
\end{lemma}

\begin{proof}
For $x^{2} - x - 1 = 0$ with roots $r_{1}, r_{2}$, Vieta's formulas
give $r_{1} + r_{2} = 1$ (negative of coefficient of $x$) and
$r_{1} r_{2} = -1$ (constant term).
\end{proof}

\begin{definition}[Forced Constants]\label{def:forced}
The axiom forces:
\begin{align}
  \chi &= 2 && \text{(Euler characteristic of $S^{2}$)},\label{eq:chi}\\
  d &= 3 && \text{(vertex degree of dodecahedron)},\label{eq:d}\\
  p &= 5 = \Delta && \text{(face degree = discriminant)}.\label{eq:p}
\end{align}
The dodecahedron $\{5,3\}$ has $V = 20$, $E = 30$, $F = 12$,
satisfying $V - E + F = \chi = 2$.
\end{definition}

\section{The Field and Its Ring of Integers}\label{sec:field}
% =============================================================================

\begin{definition}[The Number Field]\label{def:field}
$\Q(\sqrt{5}) = \{a + b\sqrt{5} : a, b \in \Q\}$ is a degree-$2$
extension of $\Q$.  Equivalently,
\begin{equation}\label{eq:field-quotient}
  \Q(\sqrt{5}) \cong \Q[x]/(x^{2} - x - 1).
\end{equation}
\end{definition}

\begin{proof}[Proof of the isomorphism]
The polynomial $f(x) = x^{2} - x - 1$ is irreducible over $\Q$
(its roots $\vp, \ps$ are irrational, as $\sqrt{5} \notin \Q$).
Therefore $\Q[x]/(f)$ is a field of degree $2$ over $\Q$.  Since
$\vp = (1 + \sqrt{5})/2 \in \Q(\sqrt{5})$ and $[\Q(\vp):\Q] = 2
= [\Q(\sqrt{5}):\Q]$, we have $\Q(\vp) = \Q(\sqrt{5})$.
The evaluation map $x \mapsto \vp$ induces the isomorphism
$\Q[x]/(x^{2} - x - 1) \cong \Q(\vp) = \Q(\sqrt{5})$.
\end{proof}

\begin{definition}[The Ring of Integers]\label{def:ring}
The ring of integers of $\Q(\sqrt{5})$ is
\begin{equation}\label{eq:ring}
  \Z[\vp] = \{a + b\vp : a, b \in \Z\}.
\end{equation}
This is the maximal order of $\Q(\sqrt{5})$, since $5 \equiv 1 \pmod{4}$
and the ring of integers of $\Q(\sqrt{D})$ with $D \equiv 1 \pmod{4}$
is $\Z[(1+\sqrt{D})/2]$.  For $D = 5$: $(1+\sqrt{5})/2 = \vp$.
\end{definition}

\begin{theorem}[$\Z[\vp]$ Is Closed Under $+$ and $\times$]\label{thm:closure}
For all $a + b\vp, \; c + d\vp \in \Z[\vp]$:
\begin{align}
  (a + b\vp) + (c + d\vp) &= (a+c) + (b+d)\vp \in \Z[\vp],
  \label{eq:add-closed}\\
  (a + b\vp)(c + d\vp) &= (ac + bd) + (ad + bc + bd)\vp \in \Z[\vp].
  \label{eq:mult-closed}
\end{align}
\end{theorem}

\begin{proof}
Addition: $(a+c) \in \Z$ and $(b+d) \in \Z$, so the sum is in $\Z[\vp]$.

Multiplication: expand
\begin{align*}
  (a + b\vp)(c + d\vp)
  &= ac + ad\vp + bc\vp + bd\vp^{2}\\
  &= ac + (ad + bc)\vp + bd(\vp + 1) && (\text{since } \vp^{2} = \vp + 1)\\
  &= (ac + bd) + (ad + bc + bd)\vp.
\end{align*}
Since $ac + bd \in \Z$ and $ad + bc + bd \in \Z$, the product is
in $\Z[\vp]$.
\end{proof}

\begin{remark}\label{rem:axiom-closure}
The closure of multiplication uses $\vp^{2} = \vp + 1$---the axiom
itself.  Without the axiom, $\vp^{2}$ would leave $\Z[\vp]$; the axiom
is precisely the relation that keeps the ring closed.
\end{remark}

\section{The Norm}\label{sec:norm}
% =============================================================================

\begin{definition}[Field Norm]\label{def:norm}
For $\alpha = a + b\vp \in \Z[\vp]$, the norm is
\begin{equation}\label{eq:norm}
  \Norm(\alpha) = \alpha \cdot \sigma(\alpha)
  = (a + b\vp)(a + b\ps)
\end{equation}
where $\sigma : \vp \mapsto \ps$ is the nontrivial automorphism of
$\Q(\sqrt{5})/\Q$.
\end{definition}

\begin{theorem}[Norm Formula]\label{thm:norm-formula}
For $\alpha = a + b\vp \in \Z[\vp]$:
\begin{equation}\label{eq:norm-explicit}
  \Norm(a + b\vp) = a^{2} + ab - b^{2}.
\end{equation}
This is always an integer.
\end{theorem}

\begin{proof}
Compute:
\begin{align*}
  (a + b\vp)(a + b\ps)
  &= a^{2} + ab\ps + ab\vp + b^{2}\vp\ps\\
  &= a^{2} + ab(\vp + \ps) + b^{2}(\vp\ps)\\
  &= a^{2} + ab(1) + b^{2}(-1) && (\text{Lemma~\ref{lem:vieta}})\\
  &= a^{2} + ab - b^{2}.
\end{align*}
Since $a, b \in \Z$, the result $a^{2} + ab - b^{2} \in \Z$.
\end{proof}

\begin{theorem}[Multiplicativity of the Norm]\label{thm:norm-mult}
For all $\alpha, \beta \in \Z[\vp]$:
\begin{equation}\label{eq:norm-mult}
  \Norm(\alpha\beta) = \Norm(\alpha) \cdot \Norm(\beta).
\end{equation}
\end{theorem}

\begin{proof}
Since $\sigma$ is a ring homomorphism:
\[
  \Norm(\alpha\beta) = (\alpha\beta)\sigma(\alpha\beta)
  = \alpha\beta \cdot \sigma(\alpha)\sigma(\beta)
  = (\alpha\sigma(\alpha))(\beta\sigma(\beta))
  = \Norm(\alpha)\Norm(\beta).
\]
\end{proof}

\begin{corollary}[Norm of $\vp$]\label{cor:norm-phi}
$\Norm(\vp) = \vp \cdot \ps = -1$.
\end{corollary}

\begin{proof}
$\vp = 0 + 1 \cdot \vp$, so $\Norm(\vp) = 0^{2} + 0 \cdot 1 - 1^{2}
= -1$.  Alternatively, $\Norm(\vp) = \vp \cdot \ps = -1$ by
Lemma~\ref{lem:vieta}.
\end{proof}

\begin{theorem}[$\Z[\vp]$ Is a Euclidean Domain]\label{thm:euclidean}
$\Z[\vp]$ is a Euclidean domain with Euclidean function $|\Norm(\cdot)|$.
Consequently, $\Z[\vp]$ is a principal ideal domain (PID) and a unique
factorization domain (UFD).
\end{theorem}

\begin{proof}
We must show: for all $\alpha, \beta \in \Z[\vp]$ with $\beta \ne 0$,
there exist $q, r \in \Z[\vp]$ such that $\alpha = q\beta + r$ with
$|\Norm(r)| < |\Norm(\beta)|$.

Consider $\gamma = \alpha/\beta \in \Q(\sqrt{5})$.  Write
$\gamma = s + t\vp$ with $s, t \in \Q$.  Choose $m, n \in \Z$
closest to $s, t$ respectively (so $|s - m| \le 1/2$ and
$|t - n| \le 1/2$).  Set $q = m + n\vp \in \Z[\vp]$ and
$r = \alpha - q\beta = (\gamma - q)\beta$.

Then $\gamma - q = (s - m) + (t - n)\vp$.  Let $u = s - m$ and
$v = t - n$, so $|u|, |v| \le 1/2$.  The norm of the ``error'' is
\[
  |\Norm(u + v\vp)| = |u^{2} + uv - v^{2}|.
\]
We bound this.  Since $|u|, |v| \le 1/2$:
\begin{align*}
  |u^{2} + uv - v^{2}|
  &\le u^{2} + |uv| + v^{2}\\
  &\le \tfrac{1}{4} + \tfrac{1}{4} + \tfrac{1}{4}
  = \tfrac{3}{4} < 1.
\end{align*}
Therefore $|\Norm(r)| = |\Norm(\gamma - q)| \cdot |\Norm(\beta)|
< 1 \cdot |\Norm(\beta)| = |\Norm(\beta)|$, as required.

Since every Euclidean domain is a PID, and every PID is a UFD, both
consequences follow.
\end{proof}

\section{The Galois Automorphism}\label{sec:galois}
% =============================================================================

\begin{definition}[The Automorphism $\sigma$]\label{def:sigma}
The Galois group $\Gal(\Q(\sqrt{5})/\Q) = \{1, \sigma\}$ has order $2$.
The nontrivial element acts by
\begin{equation}\label{eq:sigma}
  \sigma : a + b\vp \mapsto a + b\ps \qquad (a, b \in \Q).
\end{equation}
Equivalently, $\sigma(\sqrt{5}) = -\sqrt{5}$.
\end{definition}

\begin{theorem}[$\sigma$ Is an Involution Fixing $\Q$ Pointwise]\label{thm:involution}
\begin{enumerate}[nosep]
  \item $\sigma^{2} = \mathrm{id}$ (involution).
  \item $\sigma(q) = q$ for all $q \in \Q$ (fixes rationals).
  \item $\sigma(\vp) = \ps$ and $\sigma(\ps) = \vp$ (swaps roots).
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $\sigma^{2}(a + b\vp) = \sigma(a + b\ps) = a + b\vp$ (since
$\sigma(\ps) = \sigma(1 - \vp) = 1 - \ps = \vp$).

(2) For $q \in \Q$: $q = q + 0 \cdot \vp$, so $\sigma(q) = q + 0 \cdot \ps = q$.

(3) $\sigma(\vp) = 0 + 1 \cdot \ps = \ps$.  Then $\sigma(\ps) =
\sigma(1 - \vp) = 1 - \ps = \vp$.
\end{proof}

\begin{theorem}[The Galois Fixed Point $1/2$]\label{thm:fixed-point}
The element $1/2 = (\vp + \ps)/2$ is the unique element of
$\Q(\sqrt{5}) \setminus \Q$ that is ``balanced'' under $\sigma$ in the
following sense: it is the midpoint of the two roots.

More precisely, $1/2 \in \Q$ and hence $\sigma(1/2) = 1/2$.  The
fixed field of $\sigma$ is exactly $\Q$.  The value $1/2$ is
distinguished as
\begin{equation}\label{eq:midpoint}
  \frac{\vp + \ps}{2} = \frac{1}{2}.
\end{equation}
\end{theorem}

\begin{proof}
$(\vp + \ps)/2 = 1/2$ by Lemma~\ref{lem:vieta}.  Since $1/2 \in \Q$,
$\sigma(1/2) = 1/2$ by Theorem~\ref{thm:involution}(2).

By the fundamental theorem of Galois theory, the fixed field of
$\Gal(\Q(\sqrt{5})/\Q)$ is $\Q$ itself.  Every element of $\Q$ is
fixed; every element of $\Q(\sqrt{5}) \setminus \Q$ (those with
$b \ne 0$) is moved by $\sigma$.

The value $1/2$ is the unique real number equidistant from $\vp$ and $\ps$.
In the framework, this is the termination point: the machine state
$(x, y)$ with $y = (x - (x - 2k))/2 = k$ reaches the fixed point when
$k$ represents the Galois midpoint.  The fraction $1/2$ is the arithmetic
mean of the two conjugate roots, hence the unique first-degree rational
quantity determined by both roots symmetrically.
\end{proof}

\begin{proposition}[$\vp \cdot |\ps| = 1$]\label{prop:phi-psi-product}
\begin{equation}\label{eq:phi-psi-abs}
  \vp \cdot |\ps| = 1.
\end{equation}
\end{proposition}

\begin{proof}
$\vp \cdot \ps = -1$ (Lemma~\ref{lem:vieta}).  Since $\ps < 0$,
$|\ps| = -\ps = 1/\vp$.  Therefore $\vp \cdot |\ps| = \vp \cdot (1/\vp) = 1$.
\end{proof}

\section{The Exponential Embedding}\label{sec:embedding}
% =============================================================================

\begin{definition}[The Embedding]\label{def:embedding}
The exponential embedding is the map
\begin{equation}\label{eq:embedding}
  \iota : (\R, +) \to (\R_{>0}, \times), \qquad x \mapsto \vp^{x}.
\end{equation}
\end{definition}

\begin{theorem}[$\iota$ Is a Group Homomorphism]\label{thm:homomorphism}
For all $a, b \in \R$:
\begin{equation}\label{eq:homo}
  \vp^{a+b} = \vp^{a} \cdot \vp^{b}.
\end{equation}
That is, $\iota$ sends addition to multiplication.
\end{theorem}

\begin{proof}
This is the fundamental property of the exponential function.
Define $\vp^{x} = e^{x \ln \vp}$ for $x \in \R$.  Then
\[
  \vp^{a+b} = e^{(a+b)\ln\vp} = e^{a\ln\vp + b\ln\vp}
  = e^{a\ln\vp} \cdot e^{b\ln\vp} = \vp^{a} \cdot \vp^{b}.
\]
Since $\ln\vp > 0$ (as $\vp > 1$), the map $\iota$ is injective and
continuous, hence a topological group embedding of $(\R, +)$ into
$(\R_{>0}, \times)$.
\end{proof}

\begin{remark}[The Fundamental Bridge]\label{rem:bridge}
The identity $\vp^{a} \cdot \vp^{b} = \vp^{a+b}$ is what the
framework calls the ``fundamental bridge'': it converts additive
dynamics in $x$-space to multiplicative dynamics in $y$-space.
If $x_{k+1} = x_{k} + c$ (an arithmetic step), then
$y_{k+1} = \vp^{x_{k+1}} = \vp^{x_{k} + c} = y_{k} \cdot \vp^{c}$
(a geometric step).  The embedding $\iota$ transforms a linear walk
into a multiplicative cascade, with the axiom $\vp^{2} = \vp + 1$
governing the transition rule.
\end{remark}

\begin{theorem}[Exact Inversion]\label{thm:inversion}
For all $n \in \Z_{\ge 0}$:
\begin{equation}\label{eq:inversion}
  \vp^{-n} = |\ps|^{n} = \frac{1}{\vp^{n}}.
\end{equation}
No approximation is involved.
\end{theorem}

\begin{proof}
Since $|\ps| = 1/\vp$ (Proposition~\ref{prop:phi-psi-product}):
\[
  |\ps|^{n} = \left(\frac{1}{\vp}\right)^{n} = \frac{1}{\vp^{n}} = \vp^{-n}.
\]
This is an exact algebraic identity, valid for all $n \in \Z$.
\end{proof}

\begin{corollary}[Self-Inverse Dynamics]\label{cor:self-inverse}
The embedding converts forward evolution $x_{n+1} = x_{n} + c$ to
$y_{n+1} = y_{n} \cdot \vp^{c}$ and backward evolution
$x_{n-1} = x_{n} - c$ to $y_{n-1} = y_{n} \cdot \vp^{-c}
= y_{n} \cdot |\ps|^{c}$.  The inverse uses the conjugate root,
not a separate formula.
\end{corollary}

\begin{proof}
$y_{n-1} = \vp^{x_{n} - c} = \vp^{x_{n}} \cdot \vp^{-c}
= y_{n} \cdot |\ps|^{c}$, by Theorem~\ref{thm:inversion}.
\end{proof}

\section{The Machine}\label{sec:machine}
% =============================================================================

\begin{definition}[Machine State]\label{def:machine}
The machine state at step $k$ is the pair $(x_{k}, y_{k})$ where
$x_{k} \in \R$ and $y_{k} = x_{k} - 2k_{0}$ for some offset $k_{0}$.
Equivalently, $(x, y) = (n, n - 2k)$ at step $k$.

At each step, the machine makes one of two moves:
\begin{align}
  \text{Up:} &\quad (x, y) \mapsto (x+1, y+1), \label{eq:up}\\
  \text{Down:} &\quad (x, y) \mapsto (x+1, y-1). \label{eq:down}
\end{align}
Both moves increment $x$ by $1$; they differ only in $y$.
\end{definition}

\begin{proposition}[Multiplicative Machine]\label{prop:mult-machine}
Under the embedding $Y_{k} = \vp^{x_{k}}$, the machine dynamics become:
\begin{align}
  \text{Up:} &\quad Y_{k+1} = Y_{k} \cdot \vp, \label{eq:up-mult}\\
  \text{Down:} &\quad Y_{k+1} = Y_{k} \cdot \vp. \label{eq:down-mult}
\end{align}
In the $Y$-variable, both moves are identical: multiplication by $\vp$.
The distinction between ``up'' and ``down'' is carried entirely by the
second coordinate $y_{k}$.
\end{proposition}

\begin{proof}
In both cases, $x_{k+1} = x_{k} + 1$, so
$Y_{k+1} = \vp^{x_{k}+1} = \vp^{x_{k}} \cdot \vp = Y_{k} \cdot \vp$.
\end{proof}

\begin{remark}\label{rem:y-tracking}
The $y$-coordinate tracks the ``signed altitude'' of the walk.  The
embedding $\vp^{x}$ sees only the total step count $x$, not the
direction.  Information about the walk's path (its sequence of ups and
downs) is encoded in $y$ alone.  This is the separation of concerns:
$x$ counts, $y$ remembers.
\end{remark}

\begin{definition}[Full Multiplicative Update]\label{def:full-update}
Let $g(x_{k}) \in \{+1, -1\}$ denote the machine's choice at step $k$.
Then
\begin{equation}\label{eq:full-update}
  y_{k+1} = y_{k} \cdot \vp^{g(x_{k})},
\end{equation}
where $g(x_{k}) = +1$ means ``up'' and $g(x_{k}) = -1$ means ``down''
in the $y$-variable.  (Note: both choices increment $x$ by $+1$, so
$x_{k+1} = x_{k} + 1$ regardless.)

In $y$-space, the update rule is purely multiplicative:
each step multiplies by either $\vp$ or $\vp^{-1} = |\ps|$.
\end{definition}

\section{Termination}\label{sec:termination}
% =============================================================================

\begin{theorem}[Termination at Step 291]\label{thm:termination}
The multiplicative machine in $y$-space terminates at step $n = 291$
in the following sense: $|\ps|^{291}$ is smaller than the Planck
length $\ell_{\mathrm{P}} = 1.616 \times 10^{-35}$~m.

Specifically:
\begin{equation}\label{eq:termination}
  |\ps|^{291} = \vp^{-291} = e^{-291 \ln\vp}
  \approx 10^{-291 \times 0.20898} = 10^{-60.81} \approx 1.53 \times 10^{-61}.
\end{equation}
Since $\ell_{\mathrm{P}} \approx 1.616 \times 10^{-35}$ and
$1.53 \times 10^{-61} \ll 1.616 \times 10^{-35}$, the
conjugate root's power has fallen below the Planck scale.
\end{theorem}

\begin{proof}
$\ln\vp = \ln((1+\sqrt{5})/2) = 0.48121182\ldots$

$291 \times \ln\vp = 291 \times 0.48121 = 140.03\ldots$

$\vp^{-291} = e^{-140.03} = 10^{-140.03/\ln 10}
= 10^{-140.03/2.3026} = 10^{-60.81}$.

$10^{-60.81} = 10^{-61} \times 10^{0.19} = 10^{-61} \times 1.55
\approx 1.55 \times 10^{-61}$.

The Planck length is $\ell_{\mathrm{P}} = 1.616255 \times 10^{-35}$~m
(CODATA 2022~\cite{codata2022}).

Since $1.55 \times 10^{-61} < 1.616 \times 10^{-35}$, we have
$|\ps|^{291} < \ell_{\mathrm{P}}$.

\medskip
\noindent\textbf{Boundary check.}
At $n = 290$: $\vp^{-290} = \vp^{-291} \cdot \vp = 1.55 \times 10^{-61}
\times 1.618 = 2.51 \times 10^{-61}$.  Still below $\ell_{\mathrm{P}}$.

At $n = 72$: $\vp^{-72} = 10^{-72 \times 0.20898} = 10^{-15.05}
\approx 8.9 \times 10^{-16}$.  The Planck length is $10^{-35}$, so
$\vp^{-72} \gg \ell_{\mathrm{P}}$.

The critical crossover where $\vp^{-n} \approx \ell_{\mathrm{P}}$ occurs at
$n^{*} = 35\ln 10 / \ln\vp \approx 35 \times 2.3026 / 0.48121 \approx 167.4$.

At $n = 167$: $\vp^{-167} \approx 10^{-34.9} \approx 1.3 \times 10^{-35}
\approx \ell_{\mathrm{P}}$.  At $n = 168$: below $\ell_{\mathrm{P}}$.

The framework selects $n = 291$ rather than $n = 168$ because $291$
is the step at which $2\vp^{291} \ell_{\mathrm{P}} \approx 13.80$~Gly
$\approx$ the observable universe radius (see~\cite{axiom-theory}).
The termination criterion is thus bilateral: $\vp^{291}$ reaches the
cosmological scale while $|\ps|^{291}$ falls below the Planck scale.
\end{proof}

\section{The Recurrence}\label{sec:recurrence}
% =============================================================================

\begin{definition}[The Fibonacci Recurrence]\label{def:fibonacci}
The Fibonacci sequence $(F_{n})_{n \ge 0}$ is defined by
$F_{0} = 0$, $F_{1} = 1$, and
\begin{equation}\label{eq:fib-recurrence}
  F_{n} = F_{n-1} + F_{n-2} \qquad (n \ge 2).
\end{equation}
This is the canonical second-order recurrence with characteristic
polynomial $\lambda^{2} - \lambda - 1 = 0$---the axiom.
\end{definition}

\begin{theorem}[Binet's Formula]\label{thm:binet}
For all $n \ge 0$:
\begin{equation}\label{eq:binet}
  F_{n} = \frac{\vp^{n} - \ps^{n}}{\sqrt{5}}.
\end{equation}
\end{theorem}

\begin{proof}
By induction.  Base cases: $F_{0} = (\vp^{0} - \ps^{0})/\sqrt{5}
= 0/\sqrt{5} = 0$ and $F_{1} = (\vp - \ps)/\sqrt{5}
= \sqrt{5}/\sqrt{5} = 1$.

Inductive step: assume the formula holds for $F_{n-1}$ and $F_{n-2}$.
Then
\begin{align*}
  F_{n} &= F_{n-1} + F_{n-2}\\
  &= \frac{\vp^{n-1} - \ps^{n-1}}{\sqrt{5}}
    + \frac{\vp^{n-2} - \ps^{n-2}}{\sqrt{5}}\\
  &= \frac{\vp^{n-2}(\vp + 1) - \ps^{n-2}(\ps + 1)}{\sqrt{5}}\\
  &= \frac{\vp^{n-2} \cdot \vp^{2} - \ps^{n-2} \cdot \ps^{2}}{\sqrt{5}}
    && (\vp^{2} = \vp + 1,\; \ps^{2} = \ps + 1)\\
  &= \frac{\vp^{n} - \ps^{n}}{\sqrt{5}}.
\end{align*}
Note: $\ps^{2} = \ps + 1$ holds because $\ps$ is also a root of
$x^{2} - x - 1 = 0$.
\end{proof}

\begin{theorem}[$\vp$ Is the Minimal Exponential Growth Rate]\label{thm:phi-minimal}
Among all second-order recurrences $a_{n} = pa_{n-1} + qa_{n-2}$ with
$p, q \in \Z_{>0}$, the minimal growth rate (dominant characteristic
root) is $\vp$, achieved uniquely at $p = q = 1$.
\end{theorem}

\begin{proof}
The dominant root is $\lambda_{+}(p,q) = (p + \sqrt{p^{2} + 4q})/2$.
This function is strictly increasing in both $p$ and $q$
(since $\partial\lambda_{+}/\partial p = (1 + p/\sqrt{p^{2}+4q})/2 > 0$
and $\partial\lambda_{+}/\partial q = 1/\sqrt{p^{2}+4q} > 0$).

Therefore the minimum over $\Z_{>0} \times \Z_{>0}$ is at
$(p,q) = (1,1)$:
\[
  \lambda_{+}(1,1) = \frac{1 + \sqrt{5}}{2} = \vp.
\]
For any $(p,q) \ne (1,1)$ with $p, q \ge 1$, at least one of
$p > 1$ or $q > 1$ holds, giving $\lambda_{+}(p,q) > \vp$.
\end{proof}

\section{The Binary Period}\label{sec:binary}
% =============================================================================

\begin{definition}[Pisano Period]\label{def:pisano}
The Pisano period $\pi(m)$ is the period of the Fibonacci sequence
modulo $m$:
\begin{equation}\label{eq:pisano}
  \pi(m) = \min\{k \ge 1 : F_{k} \equiv 0,\; F_{k+1} \equiv 1 \pmod{m}\}.
\end{equation}
\end{definition}

\begin{theorem}[The Binary Pisano Period]\label{thm:pisano-2}
$\pi(2) = 3 = d$.
\end{theorem}

\begin{proof}
Compute the Fibonacci sequence modulo $2$:
\begin{center}
\begin{tabular}{c|cccccccc}
$n$ & 0 & 1 & 2 & 3 & 4 & 5 & 6 & 7 \\
\hline
$F_{n}$ & 0 & 1 & 1 & 2 & 3 & 5 & 8 & 13 \\
$F_{n} \bmod 2$ & 0 & 1 & 1 & 0 & 1 & 1 & 0 & 1
\end{tabular}
\end{center}
The pattern $(0, 1, 1)$ repeats with period $3$.  Formally:
$F_{3} = 2 \equiv 0$ and $F_{4} = 3 \equiv 1 \pmod{2}$, so $\pi(2) = 3$.
\end{proof}

\begin{remark}[The Binary Bridge]\label{rem:binary-bridge}
The Pisano period $\pi(2) = 3 = d$ connects the axiom to the binary
world.  The Fibonacci recurrence $a_{n} = a_{n-1} + a_{n-2}$ over
$\F_{2}$ (where addition is XOR) produces the cyclic sequence
$0, 1, 1, 0, 1, 1, \ldots$ with period $3$.  This is the vertex
degree of the dodecahedron.  The period in binary equals the
dimension.
\end{remark}

\begin{proposition}[Pisano Periods for Small Primes]\label{prop:pisano-small}
The Pisano periods for the first few primes are:
\begin{center}
\begin{tabular}{c|ccccc}
$m$ & 2 & 3 & 5 & 7 & 11 \\
\hline
$\pi(m)$ & 3 & 8 & 20 & 16 & 10
\end{tabular}
\end{center}
In particular: $\pi(2) = 3 = d$, $\pi(3) = 8 = 2^{d}$,
$\pi(5) = 20 = V$ (the vertex count of the dodecahedron).
\end{proposition}

\begin{proof}
Each is verified by direct computation of the Fibonacci sequence
modulo $m$ until the initial pair $(0, 1)$ recurs.

$\pi(2) = 3$: proved in Theorem~\ref{thm:pisano-2}.

$\pi(3)$: $F_{n} \bmod 3$: $0, 1, 1, 2, 0, 2, 2, 1, 0, 1, \ldots$
Period $8 = 2^{3} = 2^{d}$.

$\pi(5)$: $F_{n} \bmod 5$:
$0, 1, 1, 2, 3, 0, 3, 3, 1, 4, 0, 4, 4, 3, 2, 0, 2, 2, 4, 1, 0, 1, \ldots$
Period $20 = V$.
\end{proof}

\section{The Quintic Barrier}\label{sec:quintic}
% =============================================================================

\begin{theorem}[Simplicity of $A_{5}$]\label{thm:a5-simple}
The alternating group $A_{5}$ is simple: its only normal subgroups are
$\{e\}$ and $A_{5}$ itself.
\end{theorem}

\begin{proof}
$|A_{5}| = 60$.  The conjugacy classes have sizes $1, 12, 12, 15, 20$
(identity, two classes of $5$-cycles, products of two disjoint
transpositions, and $3$-cycles respectively).  A normal subgroup
$N \lhd A_{5}$ is a union of conjugacy classes containing $\{e\}$
(size $1$), and $|N|$ must divide $60$.  We check all possible unions
of $\{1\}$ with subsets of $\{12, 12, 15, 20\}$:

\begin{center}
\begin{tabular}{ll}
$1 + 12 = 13$ & does not divide 60 \\
$1 + 15 = 16$ & does not divide 60 \\
$1 + 20 = 21$ & does not divide 60 \\
$1 + 12 + 12 = 25$ & does not divide 60 \\
$1 + 12 + 15 = 28$ & does not divide 60 \\
$1 + 12 + 20 = 33$ & does not divide 60 \\
$1 + 15 + 20 = 36$ & does not divide 60 \\
$1 + 12 + 12 + 15 = 40$ & does not divide 60 \\
$1 + 12 + 12 + 20 = 45$ & does not divide 60 \\
$1 + 12 + 15 + 20 = 48$ & does not divide 60 \\
$1 + 12 + 12 + 15 + 20 = 60$ & gives $N = A_{5}$
\end{tabular}
\end{center}
No proper nontrivial union gives a divisor of $60$.  Therefore $A_{5}$
is simple.
\end{proof}

\begin{corollary}[$A_{5}$ Is Not Solvable]\label{cor:a5-not-solvable}
Since $A_{5}$ is simple and non-abelian ($|A_{5}| = 60 > 2$), its only
composition series is $A_{5} \rhd \{e\}$, with quotient
$A_{5}/\{e\} \cong A_{5}$ non-abelian.  Therefore $A_{5}$ is not
solvable.
\end{corollary}

\begin{theorem}[Klein, 1884]\label{thm:klein}
The rotation group of the regular icosahedron (equivalently, the
dodecahedron) is isomorphic to $A_{5}$.
\end{theorem}

\begin{proof}
The icosahedron has $12$ vertices and $60$ rotational symmetries forming
a group $G \le \mathrm{SO}(3)$ of order $60$.

\emph{Step 1.}  The $12$ vertices of the icosahedron can be partitioned
into $5$ inscribed tetrahedra, each using $4$ of the $12$ vertices
such that the $4$ chosen vertices form a regular
tetrahedron~\cite{klein1884,coxeter}.

\emph{Step 2.}  Each rotation $g \in G$ permutes these $5$ tetrahedra,
giving a homomorphism $\rho : G \to S_{5}$.

\emph{Step 3.}  The kernel of $\rho$ consists of rotations fixing all $5$
tetrahedra setwise.  A nontrivial rotation must move at least one vertex,
which moves at least one tetrahedron.  So $\ker\rho = \{e\}$ and
$\rho$ is injective.

\emph{Step 4.}  $\rho(G)$ is a subgroup of $S_{5}$ of order $60$.  The
unique subgroup of $S_{5}$ with index $2$ is $A_{5}$ (which also has order
$60$).  Therefore $\rho(G) = A_{5}$, and $G \cong A_{5}$.
\end{proof}

\begin{theorem}[The Quintic Barrier]\label{thm:quintic}
The general polynomial of degree $n \ge 5$ is not solvable by radicals.
The obstruction is the non-solvability of $A_{5}$, which is the rotation
group of the dodecahedron.
\end{theorem}

\begin{proof}
By Galois theory~\cite{galois1832}, a polynomial $f(x) \in \Q[x]$ is
solvable by radicals if and only if its Galois group is solvable.  The
generic polynomial of degree $5$ has Galois group $S_{5}$.  The
composition series of $S_{5}$ is $S_{5} \rhd A_{5} \rhd \{e\}$, with
quotient $S_{5}/A_{5} \cong \Z/2\Z$ (abelian) but $A_{5}/\{e\}
\cong A_{5}$ (non-abelian, by Corollary~\ref{cor:a5-not-solvable}).
Therefore $S_{5}$ is not solvable, and the general quintic admits no
solution by radicals.

For $n \ge 6$, $S_{n}$ contains $S_{5}$ as a subgroup, so $S_{n}$ is
also not solvable.

Combining with Theorem~\ref{thm:klein}: the group that blocks
solvability ($A_{5}$) is exactly the symmetry group of the dodecahedron.
The quintic barrier is dodecahedral.
\end{proof}

\begin{proposition}[Degrees $\le 4$ Are Solvable]\label{prop:solvable-small}
The symmetric groups $S_{n}$ for $n \le 4$ are all solvable:
\begin{enumerate}[nosep]
  \item $S_{1} = \{e\}$: trivially solvable.
  \item $S_{2} \cong \Z/2\Z$: abelian, hence solvable.
  \item $S_{3}$: $S_{3} \rhd A_{3} \rhd \{e\}$ with abelian quotients
    $\Z/2\Z$ and $\Z/3\Z$.
  \item $S_{4}$: $S_{4} \rhd A_{4} \rhd V_{4} \rhd \Z/2\Z \rhd \{e\}$
    with abelian quotients $\Z/2\Z, \Z/3\Z, \Z/2\Z, \Z/2\Z$.
\end{enumerate}
Therefore all polynomials of degree $\le 4$ are solvable by radicals.
\end{proposition}

\section{The Complete Chain}\label{sec:chain}
% =============================================================================

\begin{theorem}[The Axiom Chain]\label{thm:chain}
The single equation $x^{2} = x + 1$ determines, by algebraic necessity,
the following chain of structures:
\begin{enumerate}
  \item \textbf{The roots}: $\vp = (1+\sqrt{5})/2$ and
    $\ps = (1-\sqrt{5})/2$ (quadratic formula).
  \item \textbf{The field}: $\Q(\sqrt{5}) \cong \Q[x]/(x^{2}-x-1)$
    (Definition~\ref{def:field}).
  \item \textbf{The ring}: $\Z[\vp]$, a Euclidean domain
    (Theorem~\ref{thm:euclidean}).
  \item \textbf{The norm}: $\Norm(a+b\vp) = a^{2}+ab-b^{2} \in \Z$,
    multiplicative (Theorems~\ref{thm:norm-formula}
    and~\ref{thm:norm-mult}).
  \item \textbf{The involution}: $\sigma : \vp \leftrightarrow \ps$,
    fixing $\Q$ (Theorem~\ref{thm:involution}).
  \item \textbf{The fixed point}: $1/2 = (\vp+\ps)/2$
    (Theorem~\ref{thm:fixed-point}).
  \item \textbf{The embedding}: $x \mapsto \vp^{x}$, a homomorphism
    $(+) \to (\times)$ (Theorem~\ref{thm:homomorphism}).
  \item \textbf{Exact inversion}: $\vp^{-n} = |\ps|^{n}$
    (Theorem~\ref{thm:inversion}).
  \item \textbf{The recurrence}: $F_{n} = F_{n-1} + F_{n-2}$, growth
    rate $\vp$ (Theorem~\ref{thm:binet}).
  \item \textbf{Minimal growth}: $\vp$ is the smallest exponential
    growth rate among integer recurrences
    (Theorem~\ref{thm:phi-minimal}).
  \item \textbf{The binary period}: $\pi(2) = 3 = d$
    (Theorem~\ref{thm:pisano-2}).
  \item \textbf{The dodecahedron}: $\{5,3\}$, the unique Platonic solid
    with face degree $p = 5 = \Delta$.
  \item \textbf{The quintic barrier}: $A_{5} \cong \mathrm{Rot}(\{5,3\})$
    is simple and non-solvable (Theorems~\ref{thm:a5-simple},
    \ref{thm:klein}, \ref{thm:quintic}).
  \item \textbf{Termination}: $|\ps|^{291} < \ell_{\mathrm{P}}$
    (Theorem~\ref{thm:termination}).
\end{enumerate}
Each link is a proven theorem.  No conjectures, approximations, or free
parameters appear in the chain.
\end{theorem}

\begin{proof}
Each item is proved in the referenced theorem or definition.  The chain
is constructive: (1)~follows from the quadratic formula applied to the
axiom; (2)--(6) are standard algebraic number theory of $\Q(\sqrt{5})$;
(7)--(8) are properties of the exponential; (9)--(11) are combinatorics
of the Fibonacci recurrence; (12) is the unique Platonic solid
forced by the discriminant $\Delta = 5$; (13) is classical Galois
theory (Abel--Ruffini--Galois--Klein, 1824--1884); (14) is a numerical
evaluation.

No step requires input beyond the axiom $x^{2} = x + 1$ and the
definition of the $2$-sphere ($\chi = 2$).
\end{proof}
